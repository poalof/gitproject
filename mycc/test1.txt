program test1;

var
    i : integer;

function Fib(n : integer) : integer;
begin
    if n = 0 then
    begin
        Fib := 0;
    end
    else 
    begin
        if n = 1 then
        begin
            Fib := 1;
        end
        else
        begin
            Fib := Fib(n - 1) + Fib(n - 2);
        end
        ;
    end
    ;
end
;

begin
    for i := 5 to 10 do
    begin
        writeln(Fib(i));
    end
    ;
end
